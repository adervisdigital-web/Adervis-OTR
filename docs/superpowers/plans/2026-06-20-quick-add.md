  ```javascript
      function extractNameFromUrl(raw) {
          const s = String(raw || '').trim();
          if (!s) return '';
          // @handle → Telegram username
          if (s.startsWith('@')) {
              return s.slice(1).replace(/[-_.]/g, ' ').replace(/\b\w/g, c => c.toUpperCase()).trim();
          }
          try {
              const url = s.startsWith('http') ? s : 'https://' + s;
              const pathname = new URL(url).pathname;
              const slug = pathname.split('/').filter(Boolean)[0] || '';
              if (!slug || slug.match(/^\d+$/)) return ''; // skip pure numeric ids
              return slug.replace(/[-_]/g, ' ').replace(/[.]/g, ' ').replace(/\b\w/g, c => c.toUpperCase()).trim();
          } catch (e) {
              return '';
          }
      }
  ```

  ```javascript
          // ═══════════════════════════════════════════════════
          //  QUICK ADD MODAL
          // ═══════════════════════════════════════════════════
          var _qaFocusTrigger = null;
          var _qaEscHandler   = null;

          function openQuickAdd() {
              _qaFocusTrigger = document.activeElement;
              var overlay = document.getElementById('quickAddOverlay');
              if (!overlay) return;
              overlay.style.display = 'flex';
              // Reset fields
              document.getElementById('qaUrl').value = '';
              document.getElementById('qaName').value = '';
              document.getElementById('qaSegment').value = '';
              document.getElementById('qaUrlStatus').style.opacity = '0';
              document.getElementById('qaPlatformTag').style.display = 'none';
              document.getElementById('qaAutoLabel').style.display = 'none';
              document.getElementById('qaUrl').style.borderColor = 'var(--line)';
              var pbCheck = document.getElementById('qaPlaybook');
              if (pbCheck) pbCheck.checked = true;
              // Focus URL input
              setTimeout(function() {
                  var urlInput = document.getElementById('qaUrl');
                  if (urlInput) urlInput.focus();
              }, 30);
              // Esc handler
              _qaEscHandler = function(e) {
                  if (e.key === 'Escape') { e.stopPropagation(); closeQuickAdd(); }
              };
              document.addEventListener('keydown', _qaEscHandler, true); // capture phase = fires first
          }

          // Alias for mobile FAB (calls openAddModal)
          var openAddModal = openQuickAdd;

          function closeQuickAdd() {
              var overlay = document.getElementById('quickAddOverlay');
              if (overlay) overlay.style.display = 'none';
              if (_qaEscHandler) {
                  document.removeEventListener('keydown', _qaEscHandler, true);
                  _qaEscHandler = null;
              }
              if (_qaFocusTrigger && document.contains(_qaFocusTrigger)) {
                  _qaFocusTrigger.focus();
              }
              _qaFocusTrigger = null;
          }

          function onQuickAddUrlInput(val) {
              var status   = document.getElementById('qaUrlStatus');
              var platTag  = document.getElementById('qaPlatformTag');
              var platLabel= document.getElementById('qaPlatformLabel');
              var nameInput= document.getElementById('qaName');
              var autoLabel= document.getElementById('qaAutoLabel');

              if (!val.trim()) {
                  status.style.opacity = '0';
                  platTag.style.display = 'none';
                  autoLabel.style.display = 'none';
                  return;
              }

              var platformKey = detectPlatform(normalizeUrl(val));
              var platformMap = {
                  vk:   { icon: '🔵', name: 'VK' },
                  inst: { icon: '🟣', name: 'Instagram' },
                  tg:   { icon: '💬', name: 'Telegram' }
              };

              // Also handle @handle as Telegram
              var isTgHandle = val.trim().startsWith('@');
              if (isTgHandle && !platformKey) platformKey = 'tg';

              var platformInfo = platformMap[platformKey] || null;

              if (platformInfo) {
                  status.textContent = '✅';
                  status.style.opacity = '1';
                  platLabel.textContent = platformInfo.icon + ' ' + platformInfo.name;
                  platTag.style.display = 'block';
              } else {
                  status.textContent = '⚠️';
                  status.style.opacity = '1';
                  platTag.style.display = 'none';
              }

              // Auto-fill name only if field is empty
              if (!nameInput.value) {
                  var extracted = extractNameFromUrl(val);
                  if (extracted) {
                      nameInput.value = extracted;
                      autoLabel.style.display = 'inline';
                  }
              }
          }

          function submitQuickAdd() {
              var urlVal  = (document.getElementById('qaUrl').value || '').trim();
              var nameVal = (document.getElementById('qaName').value || '').trim();
              var bizType = document.getElementById('qaSegment').value || '';

              // Validation: need at least a name
              if (!nameVal && !urlVal) {
                  document.getElementById('qaUrl').style.borderColor = 'var(--danger)';
                  document.getElementById('qaUrl').focus();
                  return;
              }

              var finalName = nameVal || extractNameFromUrl(urlVal) || urlVal;
              var finalLink = normalizeUrl(urlVal);

              var newLead = {
                  id:         uid(),
                  name:       finalName,
                  link:       finalLink,
                  contact:    '',
                  status:     0,
                  bizType:    bizType,
                  updatedAt:  Date.now(),
                  notes:      '',
                  messages:   [],
                  assignedTo: currentUser ? currentUser.id : null,
                  createdBy:  currentUser ? currentUser.id : null,
                  vkPeerId:   null,
                  playbookStep: null
              };

              leads.push(newLead);
              upsertLead(newLead);
              saveDB();

              var pbCheck = document.getElementById('qaPlaybook');
              if (pbCheck && pbCheck.checked) enrollInPlaybook(newLead.id);

              closeQuickAdd();
              // Open chat for the new lead
              openTgView(null);
              selectTgLead(newLead.id);
              showToast('📥 Лид добавлен: ' + finalName);
          }
  ```
