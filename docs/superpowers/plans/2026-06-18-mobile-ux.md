# Mobile UX Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a full mobile-responsive layer to `index.html` at `≤768px` — Bottom Tab Bar, stack navigation (list → chat), keyboard handling — without touching the desktop layout.

**Architecture:** CSS media query at 768px hides the desktop header and two-panel layout; JS body-classes (`mobile-view-list` / `mobile-view-chat`) toggle visibility of the existing sidebar and chat panel, which are reflowed to full-width on mobile. No new routing, no new data layer — existing `renderTgSidebar()`, `selectTgLead()`, `renderChatView()` are reused.

**Tech Stack:** Vanilla JS, CSS Custom Properties, `visualViewport` API. All changes in `index.html` only.

  ```js
  // ═══════════════════════════════════════════════════
  //  MOBILE LAYER
  // ═══════════════════════════════════════════════════
  let mobileActiveTab = 'dialogues'; // 'dialogues' | 'table' | 'scripts'
  let _mobileView = 'list';           // 'list' | 'chat'

  function isMobile() {
    return window.innerWidth <= 768;
  }

  /** Switch bottom tab bar active tab */
  function switchMobileTab(tab) {
    mobileActiveTab = tab;
    const ids = { dialogues: 'mobTabDialogues', table: 'mobTabTable', scripts: 'mobTabScripts' };
    Object.entries(ids).forEach(function([key, id]) {
      const btn = document.getElementById(id);
      if (!btn) return;
      const active = key === tab;
      btn.style.color = active ? 'var(--primary)' : 'var(--muted)';
      btn.setAttribute('aria-selected', String(active));
    });

    if (tab === 'dialogues') {
      // Show tg-view, hide table
      const tv = document.getElementById('table-view');
      if (tv) tv.style.display = 'none';
      openTgView(null);
      setMobileView('list');
    } else if (tab === 'table') {
      // Show table view
      const tgView = document.getElementById('tg-view');
      if (tgView) tgView.style.display = 'none';
      const tv = document.getElementById('table-view');
      if (tv) { tv.style.display = 'block'; renderTable(); }
    } else if (tab === 'scripts') {
      openSettingsModal();
    }
  }

  /** Set mobile view state: 'list' or 'chat' */
  function setMobileView(view) {
    _mobileView = view;
    document.body.classList.toggle('mobile-view-list', view === 'list');
    document.body.classList.toggle('mobile-view-chat', view === 'chat');
  }

  /** Called when user taps a lead row on mobile */
  function openMobileChat(leadId) {
    selectTgLead(leadId);        // existing: sets currentChatLeadId, renders chat
    setMobileView('chat');
  }

  /** Back button in mobile chat header */
  function closeMobileChat() {
    setMobileView('list');
    currentChatLeadId = null;
    document.getElementById('tgEmpty') && (document.getElementById('tgEmpty').style.display = 'flex');
    document.getElementById('chat-view') && (document.getElementById('chat-view').style.display = 'none');
    renderTgSidebar(null);
  }
  ```

## Task 6: Keyboard Handling (visualViewport)

**Files:**
- Modify: `index.html` — new `setupMobileKeyboard()` function

- [ ] **Step 1: Add `setupMobileKeyboard()` to the mobile JS block** (from Task 3)

  Add this function inside the mobile JS block:

  ```js
  function setupMobileKeyboard() {
    if (!window.visualViewport) return;
    var tabBar = document.getElementById('mobileTabBar');
    window.visualViewport.addEventListener('resize', function() {
      var keyboardH = window.innerHeight - window.visualViewport.height;
      if (tabBar) {
        // Push tab bar up by keyboard height
        tabBar.style.transform = keyboardH > 50
          ? 'translateY(-' + keyboardH + 'px)'
          : '';
      }
      // Scroll chat feed to bottom when keyboard appears
      if (_mobileView === 'chat') {
        var feed = document.getElementById('chatFeedMain');
        if (feed) setTimeout(function() { feed.scrollTop = feed.scrollHeight; }, 50);
      }
    });
  }
  ```

  ```js
  function openMobileMoreSheet() {
    var overlay = document.getElementById('mobileMoreOverlay');
    var sheet   = document.getElementById('mobileMoreSheet');
    if (!overlay || !sheet) return;
    overlay.style.display = 'block';
    sheet.style.display   = 'block';
    // Update theme icon
    var isDark = document.documentElement.getAttribute('data-theme') !== 'light';
    var icon  = document.getElementById('mobileThemeIcon');
    var label = document.getElementById('mobileThemeLabel');
    if (icon)  icon.textContent  = isDark ? '☀️' : '🌙';
    if (label) label.textContent = isDark ? 'Светлая тема' : 'Тёмная тема';
    // Animate in
    requestAnimationFrame(function() {
      sheet.style.transform = 'translateY(0)';
    });
    document.addEventListener('keydown', _moreSheetEsc);
  }

  function closeMobileMoreSheet() {
    var overlay = document.getElementById('mobileMoreOverlay');
    var sheet   = document.getElementById('mobileMoreSheet');
    if (!sheet) return;
    sheet.style.transform = 'translateY(100%)';
    setTimeout(function() {
      if (overlay) overlay.style.display = 'none';
      if (sheet)   sheet.style.display   = 'none';
    }, 260);
    document.removeEventListener('keydown', _moreSheetEsc);
  }

  function _moreSheetEsc(e) {
    if (e.key === 'Escape') closeMobileMoreSheet();
  }
  ```
